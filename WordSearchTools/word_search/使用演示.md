## 🎮 演示3: 代码调用示例（5分钟）

### 创建测试脚本

创建新文件 `MyTest.cs`:

```csharp
using UnityEngine;
using System.Collections.Generic;
using WordSearchGenerator;

public class MyTest : MonoBehaviour
{
    void Start()
    {
        Example1_SimpleGeneration();
        Example2_AdvancedGeneration();
        Example3_LoadPuzzle();
    }
    
    // 示例1: 最简单的生成
    void Example1_SimpleGeneration()
    {
        Debug.Log("\n=== 示例1: 简单生成 ===");
        
        // 准备单词
        List<string> words = new List<string> { "HELLO", "WORLD" };
        
        // 生成谜题
        Generator gen = new Generator();
        WordSearchData puzzle = gen.GenerateWordSearch(words);
        
        // 输出
        Debug.Log(puzzle.puzzleText);
    }
    
    // 示例2: 高级配置
    void Example2_AdvancedGeneration()
    {
        Debug.Log("\n=== 示例2: 高级配置 ===");
        
        List<string> words = new List<string> 
        { 
            "UNITY", "CSHARP", "GAME", "CODE" 
        };
        
        Generator gen = new Generator(() => 
        {
            Debug.Log("生成中...");
        });
        
        WordSearchData puzzle = gen.GenerateWordSearch(
            words,
            Constants.ALL_DIRECTIONS,           // 8方向
            4,                                  // 尺寸因子
            Constants.INTERSECT_BIAS_PREFER     // 偏好交叉
        );
        
        Debug.Log($"拼图尺寸: {puzzle.dimension}x{puzzle.dimension}");
        Debug.Log(puzzle.puzzleText);
        
        // 保存
        FileManager.SavePuzzleAsJson(puzzle);
    }
    
    // 示例3: 加载谜题
    void Example3_LoadPuzzle()
    {
        Debug.Log("\n=== 示例3: 加载谜题 ===");
        
        string fileName = FileManager.GetMostRecentPuzzle();
        
        if (!string.IsNullOrEmpty(fileName))
        {
            WordSearchData puzzle = FileManager.LoadPuzzleFromJson(fileName);
            
            Debug.Log($"加载: {fileName}");
            Debug.Log($"创建时间: {puzzle.createTime}");
            Debug.Log($"单词数: {puzzle.words.Count}");
        }
        else
        {
            Debug.Log("没有找到已保存的谜题");
        }
    }
}
```

## 🎮 演示6: 不同配置对比

### 配置1: 简单模式 + 小尺寸

```csharp
words = { "CAT", "DOG", "BIRD" };
directions = EASY_DIRECTIONS;  // 4方向
sizeFactor = 2;                // 小尺寸
intersectBias = RANDOM;

结果: 6x6 拼图（紧凑）
```

### 配置2: 困难模式 + 大尺寸

```csharp
words = { "CAT", "DOG", "BIRD" };
directions = ALL_DIRECTIONS;   // 8方向
sizeFactor = 6;                // 大尺寸
intersectBias = AVOID;         // 避免交叉

结果: 10x10 拼图（宽松）
```

### 配置3: 偏好交叉

```csharp
words = { "HELLO", "HELP", "HELL" };
directions = EASY_DIRECTIONS;
sizeFactor = 3;
intersectBias = PREFER;        // 偏好交叉

结果: 单词会尽量重叠（如HEL共享）
```

## 🎮 演示10: 文件管理演示

### 操作流程

```csharp
// 1. 生成并保存3个谜题
for (int i = 0; i < 3; i++)
{
    var puzzle = gen.GenerateWordSearch(words);
    FileManager.SavePuzzleAsJson(puzzle);
}

// 2. 获取所有谜题
List<string> files = FileManager.GetAllSavedPuzzles();
Debug.Log($"找到 {files.Count} 个谜题");

// 3. 加载最新的
string latest = FileManager.GetMostRecentPuzzle();
WordSearchData latestPuzzle = FileManager.LoadPuzzleFromJson(latest);

// 4. 删除旧的
string oldest = files[files.Count - 1];
FileManager.DeletePuzzle(oldest);
```
