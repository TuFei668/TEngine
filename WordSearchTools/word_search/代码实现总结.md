// 计算拼图维度
public static int GetPuzzleDimension(List<string> words, int sizeFactor)
{
    int wordLetterTotal = words.Sum(word => word.Length);
    int fromLetterCount = (int)Math.Sqrt(wordLetterTotal * sizeFactor);
    int fromLongestWord = words.Max(word => word.Length);
    return Math.Max(fromLetterCount, fromLongestWord);
}

##### 核心生成算法
```csharp
public WordSearchData GenerateWordSearch(
    List<string> words,
    Vector2Int[] directions = null,
    int? sizeFactor = null,
    int? intersectBias = null)
{
    // 1. 初始化
    this.words = words.Distinct().ToList();  // 去重
    this.dim = GetPuzzleDimension(this.words, this.sizeFactor);
    ResetGenerationData();
    
    // 2. 回溯算法主循环
    isHalted = false;
    while (currentIndex < this.words.Count && !isHalted)
    {
        if (没有可用位置)
        {
            // 回溯
            currentIndex--;
            if (currentIndex < 0)
            {
                // 拼图太小，增大尺寸
                dim++;
                ResetGenerationData();
            }
            else
            {
                // 恢复上一个状态
                table = tableHistory[currentIndex];
                tableHistory.RemoveAt(currentIndex);
                CurrentWorkablePositions.RemoveAt(0);
            }
        }
        else
        {
            // 保存状态并放置单词
            tableHistory.Add(CloneTable(table));
            PlaceWord();
            currentIndex++;
        }
        
        ProgressStep();
    }
    
    // 3. 渲染和返回
    return CreateWordSearchData();
}
```

##### 渲染方法
```csharp
private string RenderPuzzle(char[,] grid, int dimension, 
                            bool answerKey, List<WordPosition> wordPositions)
{
    // 1. 克隆网格
    char[,] renderTable = CloneTable(grid);
    
    // 2. 填充
    if (answerKey)
    {
        // 答案模式：点填充 + 小写 + 首字母大写
        FillWithDots(renderTable);
        CapitalizeFirstLetters(renderTable, wordPositions);
    }
    else
    {
        // 拼图模式：随机字母填充
        FillWithRandom(renderTable);
    }
    
    // 3. 旋转和翻转（模拟numpy.rot90和fliplr）
    char[,] displayTable = RotateAndFlip(renderTable, dimension);
    
    // 4. 转换为字符串
    return ConvertToString(displayTable);
}
```

## 🔍 关键实现细节

### 1. NumPy替代方案

Python使用NumPy进行高效数组操作，C#通过以下方式替代：

```csharp
// Python: np.empty((dim, dim), dtype=str)
// C#:
char[,] table = new char[dim, dim];

// Python: puzzle[indices] == wordarr
// C#:
for (int i = 0; i < wordLen; i++)
{
    if (puzzle[xIndices[i], yIndices[i]] != word[i])
        return false;
}

// Python: np.logical_or(a, b)
// C#:
bool result = (condition1 || condition2);
```

### 2. 回溯算法核心逻辑

```
开始
  ↓
初始化 (dim, table, positions)
  ↓
currentIndex = 0
  ↓
┌──────────────────┐
│ 获取可用位置      │
└────┬─────────────┘
     │
┌────┴────┐
│有位置？  │
└────┬────┘
     │
 是 ←┴→ 否
 │       │
 │       ↓
 │   回溯 (index--)
 │       │
 │   ┌───┴───┐
 │   │index<0?│
 │   └───┬───┘
 │       │
 │   是 ←┴→ 否
 │   │       │
 │   ↓       ↓
 │  扩容   恢复状态
 │   │       │
 │   └───┬───┘
 ↓       │
保存状态  │
 ↓       │
放置单词  │
 ↓       │
index++   │
 │       │
 └───┬───┘
     │

 ↓ 是
返回结果
```

### 3. 序列化处理

Unity的JsonUtility不支持多维数组，需要转换：

```csharp
// 序列化前：char[,] → string
public void GridToString()
{
    StringBuilder sb = new StringBuilder();
    for (int y = 0; y < dimension; y++)
    {
        for (int x = 0; x < dimension; x++)
        {
            sb.Append(grid[x, y]);
        }
        if (y < dimension - 1) sb.Append('|');
    }
    gridString = sb.ToString();
}

// 反序列化后：string → char[,]
public void StringToGrid()
{
    grid = new char[dimension, dimension];
    string[] rows = gridString.Split('|');
    for (int y = 0; y < dimension; y++)
    {
        for (int x = 0; x < dimension; x++)
        {
            grid[x, y] = rows[y][x];
        }
    }
}
```
